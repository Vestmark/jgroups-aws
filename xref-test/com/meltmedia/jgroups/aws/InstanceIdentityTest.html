19  public class InstanceIdentityTest {
20    @Test
21    public void fromResponse() throws Exception {
22      final HttpClient client = mock(HttpClient.class);
23      final HttpResponse response = mock(HttpResponse.class);
24      final StatusLine statusLine = mock(StatusLine.class);
25      final HttpEntity responseEntity = mock(HttpEntity.class);
26  
27      when(responseEntity.getContent()).thenReturn(new ByteArrayInputStream(Resources.toByteArray(getResource("instance-identity.json"))));
28      when(responseEntity.getContentLength()).thenReturn(-1L);
29      when(statusLine.getStatusCode()).thenReturn(HttpStatus.SC_OK);
30      when(response.getStatusLine()).thenReturn(statusLine);
31      when(response.getEntity()).thenReturn(responseEntity);
32      when(client.execute(any())).thenReturn(response);
33  
34      InstanceIdentity instanceIdentity = InstanceIdentity.getIdentity(client);
35  
36      assertEquals("us-west-2b", instanceIdentity.availabilityZone);
37      assertEquals("10.158.112.84", instanceIdentity.privateIp);
38      assertEquals("i-1234567890abcdef0", instanceIdentity.instanceId);
39      assertEquals("t2.micro", instanceIdentity.instanceType);
40      assertEquals("ami-5fb8c835", instanceIdentity.imageId);
41      assertEquals("x86_64", instanceIdentity.architecture);
42      assertEquals("us-west-2", instanceIdentity.region);
43    }
