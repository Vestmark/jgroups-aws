8   import static com.meltmedia.jgroups.aws.Mocks.ec2Mock;
9   import static org.junit.Assert.*;
10  
11  public class TagUtilsTest {
12    private static InstanceIdentity instanceIdentity = new InstanceIdentity(
13        "zone",
14        "1.2.3.4",
15        "instance_id",
16        "instance_type",
17        "image_id",
18        "architecture",
19        "region");
20  
21    @Test
22    public void canHandleNullTagString() {
23      final TagsUtils tagsUtils = new TagsUtils(ec2Mock(), instanceIdentity, null);
24      assertFalse(tagsUtils.getAwsTagNames().isPresent());
25    }
26  
27    @Test
28    public void canHandleEmptyTagString() {
29      final TagsUtils tagsUtils = new TagsUtils(ec2Mock(), instanceIdentity, "");
30      assertFalse(tagsUtils.getAwsTagNames().isPresent());
31    }
32  
33    @Test
34    public void willParseConfiguredTags() {
35      final TagsUtils tagsUtils = new TagsUtils(ec2Mock(), instanceIdentity, "configuredTag1, configuredTag2");
36      assertTrue(tagsUtils.getAwsTagNames().isPresent());
37      tagsUtils.getAwsTagNames().ifPresent(tags -> {
38        assertEquals(2, tags.size());
39        assertArrayEquals(new String[]{"configuredTag1", "configuredTag2"}, tags.toArray());
40      });
41    }
42  
43    @Test
44    public void canGetInstanceTags() {
45      final Tag instanceTag1 = new Tag("instanceTag1", "value1");
46      final Tag instanceTag2 = new Tag("instanceTag2", "value2");
47  
48      final TagsUtils tagsUtils = new TagsUtils(ec2Mock(instanceTag1, instanceTag2), instanceIdentity, null);
49      final List<Tag> tags = tagsUtils.getInstanceTags();
50  
51      assertEquals(2, tags.size());
52      Tag tag1 = tags.get(0);
53      Tag tag2 = tags.get(1);
54  
55      assertEquals("instanceTag1", tag1.getKey());
56      assertEquals("value1", tag1.getValue());
57      assertEquals("instanceTag2", tag2.getKey());
58      assertEquals("value2", tag2.getValue());
59    }
60  }
