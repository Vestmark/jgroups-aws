21  @RunWith(Parameterized.class)
22  public class PortRangeTest {
23  
24    @Parameters(name = "{0}")
25    public static Collection<Object[]> parameters() throws UnknownHostException {
26      return Arrays.asList(new Object[][]{
27          {
28              "range 0",
29              new IPAddressUtils(7800, 0),
30              matchClusterPorts(7800)
31          },
32          {
33              "range 1",
34              new IPAddressUtils(7800, 1),
35              matchClusterPorts(7800, 7801)
36          },
37          {
38              "range 2",
39              new IPAddressUtils(7800, 2),
40              matchClusterPorts(7800, 7801, 7802)
41          },
42          {
43              "different range start",
44              new IPAddressUtils(7803, 3),
45              matchClusterPorts(7803, 7804, 7805, 7806)
46          },
47          {
48              "default port and range",
49              new IPAddressUtils(new AWS_PING().port_number, new AWS_PING().port_range),
50              Matchers.hasSize(51)
51          }
52      });
53    }
54  
55    private final IPAddressUtils ipAddressUtils;
56    private Matcher<List<IpAddress>> matcher;
57  
58    public PortRangeTest(String name, IPAddressUtils ipAddressUtils, Matcher<List<IpAddress>> matcher) {
59      this.ipAddressUtils = ipAddressUtils;
60      this.matcher = matcher;
61    }
62  
63    @Test
64    public void correctClusterMembers() {
65      assertThat(ipAddressUtils.expandClusterMemberPorts(Collections.singletonList("localhost")), matcher);
66    }
67  
68    public static Matcher<Iterable<? extends PhysicalAddress>> matchClusterPorts(int... ports) throws UnknownHostException {
69      PhysicalAddress[] addresses = new PhysicalAddress[ports.length];
70      for (int i = 0; i < ports.length; i++) {
71        addresses[i] = new IpAddress("localhost", ports[i]);
72      }
73      return containsInAnyOrder(addresses);
74    }
