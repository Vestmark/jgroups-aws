7   import java.util.List;
8   import java.util.stream.Collectors;
9   import java.util.stream.IntStream;
10  
11  import static java.util.stream.Stream.empty;
12  import static java.util.stream.Stream.of;
13  
14  public class IPAddressUtils {
15    private static Log log = LogFactory.getLog(AWS_PING.class);
16  
17    private final int portNumber;
18    private final int portRange;
19  
20    public IPAddressUtils(int portNumber, int portRange) {
21      this.portNumber = portNumber;
22      this.portRange = portRange;
23    }
24  
25    public List<IpAddress> expandClusterMemberPorts(final List<String> privateIpAddresses) {
26      return privateIpAddresses.stream()
27          .flatMap(address -> IntStream.range(portNumber, portNumber + portRange + 1)
28              .mapToObj(port -> new AddressAndPort(address, port)))
29          .flatMap(addressAndPort -> {
30            try {
31              return of(new IpAddress(addressAndPort.address, addressAndPort.port));
32            } catch (Exception e) {
33              log.warn("failed to create ip address", e);
34              return empty();
35            }
36          })
37          .collect(Collectors.toList());
38    }
39  
40    private static class AddressAndPort {
41      public final String address;
42      public final int port;
43  
44      private AddressAndPort(final String address, final int port) {
45        this.address = address;
46        this.port = port;
47      }
48    }
49  }
