1   package com.meltmedia.jgroups.aws;
2   
3   import com.amazonaws.services.ec2.model.Filter;
4   import org.jgroups.util.Tuple;
5   
6   import java.util.Arrays;
7   import java.util.Collections;
8   import java.util.List;
9   import java.util.Optional;
10  import java.util.stream.Collectors;
11  import java.util.stream.Stream;
12  
13  public class FilterUtils {
14    private final TagsUtils tagsUtils;
15    @SuppressWarnings("OptionalUsedAsFieldOrParameterType")
16    private final Optional<List<Filter>> awsFilters;
17  
18    @SuppressWarnings("OptionalUsedAsFieldOrParameterType")
19    public FilterUtils(final String configuredFilters, final TagsUtils tagsUtils) {
20      this.tagsUtils = tagsUtils;
21      this.awsFilters = Optional.ofNullable(configuredFilters)
22          .map(String::trim)
23          .filter(filters -> !filters.isEmpty())
24          .map(FilterUtils::parseFilters);
25    }
26  
27    /**
28     * @return an optional list of parsed tag names
29     */
30    public Optional<List<Filter>> getAwsFilters() {
31      return awsFilters;
32    }
33  
34    /**
35     * Takes the list of configured tag names and compares it with tagsUtils on the ec2 instance.
36     * FilterUtils (tag:key=value) will be created for all matches.
37     *
38     * @return a list of filters for instances that share common tags with
39     *   this instance.
40     */
41    @SuppressWarnings("OptionalUsedAsFieldOrParameterType")
42    public List<Filter> instanceTagNamesToFilters() {
43      return tagsUtils.getAwsTagNames().map(Stream::of).orElseGet(Stream::empty)
44          .map(tagNames -> new Tuple<>(tagNames, tagsUtils.getInstanceTags()))
45          .flatMap(namesAndTags -> namesAndTags.getVal2()
46              .stream()
47              .filter(tag -> namesAndTags.getVal1().contains(tag.getKey())))
48          .map(tag -> new Filter("tag:" + tag.getKey(), Collections.singletonList(tag.getValue())))
49          .collect(Collectors.toList());
50    }
51  
52    /**
53     * Parses a filter string into a list of Filter objects that is suitable for
54     * the AWS describeInstances method call.
55     * <p>
56     * <h3>Format:</h3>
57     * <p>
58     * <blockquote>
59     * <p>
60     * <pre>
61     *   FILTERS ::= &lt;FILTER&gt; ( ';' &lt;FILTER&gt; )*
62     *   FILTER ::= &lt;NAME&gt; '=' &lt;VALUE&gt; (',' &lt;VALUE&gt;)*
63     * </pre>
64     * <p>
65     * </blockquote>
66     * </p>
67     *
68     * @param filters the filter string to parse.
69     * @return the list of filters that represent the filter string.
70     */
71    private static List<Filter> parseFilters(String filters) {
72      return Arrays.stream(filters.split("\\s*;\\s*"))
73          .map(String::trim)
74          .filter(s -> !s.isEmpty())
75          .map(FilterUtils::splitToTuple)
76          .map(keyAndValue -> new Filter(keyAndValue.getVal1(), splitValues(keyAndValue.getVal2())))
77          .collect(Collectors.toList());
78    }
79  
80    private static Tuple<String, String> splitToTuple(String filter) {
81      final String[] keyValues = filter.split("\\s*=\\s*");
82      if (keyValues.length != 2 || keyValues[0].length() == 0 || keyValues[1].length() == 0) {
83        throw new IllegalArgumentException("Could not process key value pair '" + filter + "'");
84      }
85      return new Tuple<>(keyValues[0].trim(), keyValues[1].trim());
86    }
87  
88    private static List<String> splitValues(String filterValues) {
89      return Arrays.stream(filterValues.split("\\s*,\\s*"))
90          .map(String::trim)
91          .collect(Collectors.toList());
92    }
93  }
