64  /**
65   * <p>
66   * A discovery protocol that uses the AWS EC2 API to find cluster members.
67   * Membership can be determined by a general filter and/or by nodes that have
68   * similar tags. This discovery protocol is designed to work with the TCP
69   * protocol.
70   * </p>
71   *
72   * <h3>Requirements</h3>
73   * <ul>
74   *   <li>The EC2 instances must be in the same region.</li>
75   *   <li>The "ec2:Describe*" action must be accessable using either IAM credentials or an IAM instance profile.</li>
76   *   <li>The security rules must allow TCP communication between the nodes that are discovered.</li>
77   * </ul>
78   *
79   * <h3>Tag Matching</h3>
80   * <p>
81   * To use the tag matching feature, use the tags attribute to specify a comma delimited list of tags.  All of the nodes
82   * with matching values for these tags will be discovered.
83   * </p>
84   *
85   * <blockquote>
86   * <pre>
87   * &lt;com.meltmedia.jgroups.aws.AWS_PING
88   *   timeout="3000"
89   *   port_number="7800"
90   *   tags="Type,Environment"
91   *   access_key="YOUR_AWS_ACCESS_KEY"
92   *   secret_key="YOUR_AWS_SECRET_KEY"/&gt;
93   * </pre>
94   * </blockquote>
95   *
96   * <h3>EC2 Filters</h3>
97   * <p>
98   * To use EC2's filtering feature to discover nodes, specify the filters attribute.  The format for this attribute is:
99   * </p>
100  *
101  * <blockquote>
102  * <pre>
103  * FILTERS ::= &lt;FILTER&gt; (';' &lt;FILTER&gt;)*
104  * FILTER  ::= &lt;NAME&gt; '=' &lt;VALUE&gt; (',' &lt;VALUE&gt;)*
105  * </pre>
106  * </blockquote>
107  *
108  * <p>
109  * EC2 instances that match all of the supplied filters will be returned.  For example,
110  * if you wanted to cluster with all of the running, small instances in your account, you could specify:
111  * </p>
112  * <blockquote>
113  * <pre>
114  * &lt;com.meltmedia.jgroups.aws.AWS_PING
115  *   timeout="3000"
116  *   port_number="7800"
117  *   filters="instance-state-name=running;instance-type=m1.small"
118  *   access_key="YOUR_AWS_ACCESS_KEY"
119  *   secret_key="YOUR_AWS_SECRET_KEY"/&gt;
120  * </pre>
121  * </blockquote>
122  *
123  * <h3>IAM Instance Profiles</h3>
124  * <p>
125  * Starting with version 1.1.0, instance profiles are supported by AWS_PING.  To use the instance profile associated with
126  * an EC2 instance, simply omit the access_key and secret_key attributes:
127  * </p>
128  * <blockquote>
129  * <pre>
130  * &lt;com.meltmedia.jgroups.aws.AWS_PING
131  *   timeout="3000"
132  *   port_number="7800"
133  *   tags="Type,Environment"/&gt;
134  * </pre>
135  * </blockquote>
136  *
137  * <h3>References</h3>
138  * <ul>
139  *   <li><a href="http://docs.amazonwebservices.com/AWSEC2/latest/UserGuide/Using_Filtering.html">EC2 Using Filtering</a></li>
140  *   <li><a href="http://docs.amazonwebservices.com/AWSEC2/latest/CommandLineReference/ApiReference-cmd-DescribeInstances.html">EC2 Describe Instances</a></li>
141  *   <li><a href="http://docs.amazonwebservices.com/AWSEC2/latest/UserGuide/AESDG-chapter-instancedata.html">EC2 Instance Metadata</a></li>
142  * </ul>
143  *
144  * @author Christian Trimble
145  * @author John McEntire

310   /**
311    * Fetches all of the cluster members found on EC2. The host portion of the
312    * addresses are the private ip addresses of the matching nodes. The port
313    * numbers of the addresses are set to the port number plus all the ports in
314    * the range after that specified on this protocol.
315    *
316    * @param members
317    * @param initial_discovery
318    * @param responses
319    */
320   @Override
321   protected void findMembers(List<Address> members, boolean initial_discovery, Responses responses) {
322     PhysicalAddress physical_addr = (PhysicalAddress) down(new Event(Event.GET_PHYSICAL_ADDRESS, local_addr));
323     PingData data = new PingData(local_addr, false, NameCache.get(local_addr), physical_addr);
324     PingHeader hdr = new PingHeader(PingHeader.GET_MBRS_REQ).clusterName(cluster_name);
325 
326     List<PhysicalAddress> clusterMembers =
327       expandClusterMemberPorts(getPrivateIpAddresses());
328 
329     for (final PhysicalAddress addr : clusterMembers) {
330       if (physical_addr != null && addr.equals(physical_addr))
331         continue;
332 
333       final Message msg = new Message(addr).setFlag(Message.Flag.INTERNAL, Message.Flag.DONT_BUNDLE, Message.Flag.OOB)
334           .putHeader(this.id, hdr).setBuffer(marshal(data));
335 
336       if (async_discovery_use_separate_thread_per_request) {
337         timer.execute(new Runnable() {
338           public void run() {
339             log.trace("%s: sending discovery request to %s", local_addr, msg.getDest());
340             down_prot.down(msg);
341           }
342         });
343       } else {
344         log.trace("%s: sending discovery request to %s", local_addr, msg.getDest());
345         down_prot.down(msg);
346       }
347     }
348   }

350   protected List<PhysicalAddress> expandClusterMemberPorts( List<String> privateIpAddresses) {
351     List<PhysicalAddress> clusterMembers = new ArrayList<PhysicalAddress>();
352     for (String privateIpAddress : privateIpAddresses) {
353       for (int i = port_number; i <= port_number + port_range; i++) {
354         try {
355           clusterMembers.add(new IpAddress(privateIpAddress, i));
356         } catch (UnknownHostException e) {
357           log.warn("Could not create an IpAddress for " + privateIpAddress + ":" + i);
358         }
359       }
360     }
361     return clusterMembers;
362   }

364   /**
365    * Gets the list of private IP addresses found in AWS based on the filters and
366    * tag names defined.
367    *
368    * @return the list of private IP addresses found on AWS.
369    */
370   protected List<String> getPrivateIpAddresses() {
371     List<String> result = new ArrayList<String>();
372 
373     List<Filter> filters = new ArrayList<Filter>();
374 
375     // if there are aws tags defined, then look them up and create filters.
376     if (awsTagNames != null) {
377       Collection<Tag> instanceTags = getInstanceTags(ec2, instanceId);
378       for (Tag instanceTag : instanceTags) {
379         if (awsTagNames.contains(instanceTag.getKey())) {
380           filters.add(new Filter("tag:" + instanceTag.getKey(), Arrays
381               .asList(instanceTag.getValue())));
382         }
383       }
384 
385     }
386 
387     // if there are aws filters defined, then add them to the list.
388     if (awsFilters != null) {
389       filters.addAll(awsFilters);
390     }
391     DescribeInstancesRequest request = new DescribeInstancesRequest()
392         .withFilters(filters);
393     if (log.isDebugEnabled()) {
394       log.debug("Describing AWS instances with the following filters ["
395           + filters + "]");
396       log.debug("Making AWS Request {" + request + "}");
397     }
398     // NOTE: the reservations group nodes together by when they were started. We
399     // need to dig through all of the reservations.
400     DescribeInstancesResult filterResult = ec2.describeInstances(request);
401     for (Reservation reservation : filterResult.getReservations()) {
402       for (Instance instance : reservation.getInstances()) {
403         result.add(instance.getPrivateIpAddress());
404       }
405     }
406     if (log.isDebugEnabled()) {
407       log.debug("Instances found [" + result + "]");
408     }
409     return result;
410   }

430   /**
431    * Parses a filter string into a list of Filter objects that is suitable for
432    * the AWS describeInstances method call.
433    *
434    * <h3>Format:</h3>
435    * <p>
436    * <blockquote>
437    *
438    * <pre>
439    *   FILTERS ::= &lt;FILTER&gt; ( ';' &lt;FILTER&gt; )*
440    *   FILTER ::= &lt;NAME&gt; '=' &lt;VALUE&gt; (',' &lt;VALUE&gt;)*
441    * </pre>
442    *
443    * </blockquote>
444    * </p>
445    *
446    * @param filters
447    *          the filter string to parse.
448    * @return the list of filters that represent the filter string.
449    */
450   static List<Filter> parseFilters(String filters) {
451     List<Filter> awsFilters = new ArrayList<Filter>();
452 
453     for (String filterString : filters.split("\\s*;\\s*")) {
454       // clean up the filter, moving on if it is empty.
455       String trimmed = filterString.trim();
456       if (trimmed.length() == 0)
457         continue;
458 
459       // isolate the key and the values, failing if there is a problem.
460       String[] keyValues = trimmed.split("\\s*=\\s*");
461       if (keyValues.length != 2 || keyValues[0].length() == 0
462           || keyValues[1].length() == 0)
463         throw new IllegalArgumentException("Could not process key value pair '"
464             + filterString + "'");
465 
466       // create the filter and add it to the list.
467       awsFilters.add(new Filter().withName(keyValues[0]).withValues(
468           keyValues[1].split("\\s*,\\s*")));
469     }
470     return awsFilters;
471   }

505   /**
506    * Returns all of the tags defined on the EC2 instance with the specified
507    * instanceId.
508    *
509    * @param ec2
510    *          the client to use when accessing Amazon.
511    * @param instanceId
512    *          the id of the instance to search for tags.
513    * @return a list of the Tag objects that were found on the instance.
514    */
515   public static List<Tag> getInstanceTags(AmazonEC2 ec2, String instanceId) {
516     List<Tag> tags = new ArrayList<Tag>();
517     DescribeInstancesResult response = ec2
518         .describeInstances(new DescribeInstancesRequest()
519             .withInstanceIds(Arrays.asList(instanceId)));
520     for (Reservation res : response.getReservations()) {
521       for (Instance inst : res.getInstances()) {
522         List<Tag> instanceTags = inst.getTags();
523         if (instanceTags != null && instanceTags.size() > 0) {
524           tags.addAll(instanceTags);
525         }
526       }
527     }
528     return tags;
529   }

537   private class AWSFaultLogger implements Unmarshaller<AmazonServiceException, Node>, RequestHandler {
538     private final ThreadLocal<Request<?>> request = new ThreadLocal<Request<?>>();

540     @Override
541     public AmazonServiceException unmarshall(Node node) throws Exception {
542       try {
543         javax.xml.transform.TransformerFactory tfactory = javax.xml.transform.TransformerFactory.newInstance();
544         javax.xml.transform.Transformer xform = tfactory.newTransformer();
545 
546         javax.xml.transform.Source src = new javax.xml.transform.dom.DOMSource(node);
547         java.io.StringWriter writer = new java.io.StringWriter();
548         javax.xml.transform.Result result = new javax.xml.transform.stream.StreamResult(writer);
549         xform.transform(src, result);
550         log.error("AWS Exception: [" + writer.toString()+"] For request ["+request.get()+"]");
551       } catch(Throwable t) {
552         log.debug("Failed to log xml soap fault message.", t);
553       } finally {
554         request.remove();
555       }
556       return null;
557     }

561       this.request.remove();
562     }

564     @Override
565     public void afterResponse(Request<?> request, Object obj, TimingInfo timing) {
566       this.request.remove();
567     }

569     @Override
570     public void beforeRequest(Request<?> request) {
571       this.request.set(request);

577    */
578   static AWSCredentialsProvider loadCredentialsProvider( String credentialProviderClass, Class<?> jgroupsClass, Log log )
579     throws Exception
580   {
581     Class<?> credsProviderClazz = null;
582     try {
583       credsProviderClazz = Util.loadClass(credentialProviderClass, jgroupsClass);
584     }
585     catch( ClassNotFoundException e ) {
586       throw new Exception("unable to load credentials provider class " + credentialProviderClass);
587     }
588

590     try {
591       awsCredentialsProvider = (AWSCredentialsProvider) credsProviderClazz.newInstance();
592     }
593     catch( InstantiationException e ) {
594       log.error("an instance of " + credentialProviderClass + " could not be created. Please check that it implements" +
595                 " interface AWSCredentialsProvider and that is has a public empty constructor !");
596       throw e;
597     }
598     return awsCredentialsProvider;
